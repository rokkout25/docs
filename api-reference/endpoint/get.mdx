# Example: Get SoundCloud Track

> Get SoundCloud track information

## OpenAPI

```yaml
openapi: 3.1.0
info:
  title: OpenAPI SoundCloud API
  description: API to fetch SoundCloud track information
  license:
    name: MIT
  version: 1.0.0
servers:
  - url: http://sandbox.mintlify.com
security:
  - bearerAuth: []
paths:
  /soundcloud/track:
    get:
      summary: Track
      description: Get SoundCloud track information
      operationId: SoundcloudTrackTrack
      parameters:
        - name: url
          in: query
          required: true
          deprecated: false
          allowEmptyValue: false
          allowReserved: false
          schema:
            type: string
      responses:
        '200':
          description: Request fulfilled, document follows
          content:
            application/json:
              schema:
                type: object
        '400':
          description: Bad request syntax or unsupported method
          content:
            application/json:
              schema:
                type: object
                required:
                  - status_code
                  - detail
                properties:
                  status_code:
                    type: integer
                  detail:
                    type: string
                  extra:
                    type: [null, object, array]
                    additionalProperties: true
                description: Validation Exception
                examples:
                  - status_code: 400
                    detail: Bad Request
                    extra: {}
components:
  securitySchemes:
    bearerAuth:
      type: http
      scheme: bearer
